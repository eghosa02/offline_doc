\chapter{Processo di generazione dei verbali}
\label{chap:verbali}

\section{Obiettivo}
Descrivere il processo per generare verbali delle chiamate.

\section{Struttura del verbale}
\begin{itemize}
  \item data e orario;
  \item partecipanti;
  \item argomenti;
  \item decisioni;
  \item azioni;
  \item punti aperti.
\end{itemize}

\section{Procedura}
\begin{enumerate}
  \item prendere appunti;
  \item organizzare le informazioni;
  \item scrivere il verbale;
  \item salvarlo nella repository;
  \item condividerlo.
\end{enumerate}

\section{Aggiornamento continuo}
Creare un verbale dopo ogni chiamata.
